963#define AGIN2OUT(inedge) ((inedge) - 1)
964#define AGOUT2IN(outedge) \
965 ((outedge) + 1)
966#define AGOPP(e) ((AGTYPE(e) == AGINEDGE) ? AGIN2OUT(e) : AGOUT2IN(e))
967#define AGMKOUT(e) (AGTYPE(e) == AGOUTEDGE ? (e) : AGIN2OUT(e))
968#define AGMKIN(e) (AGTYPE(e) == AGINEDGE ? (e) : AGOUT2IN(e))
969#define AGTAIL(e) (AGMKIN(e)->node)
970#define AGHEAD(e) (AGMKOUT(e)->node)
971#define AGEQEDGE(e, f) (AGMKOUT(e) == AGMKOUT(f))
